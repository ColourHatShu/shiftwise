describe('Phase 8 Shift Compliance & Assignment Integration', () => {
  let agency, coordinator, shift, workers;

  beforeAll(async () => {
    // Setup: Create test agency, users, workers, compliance docs
    agency = await createTestAgency();
    coordinator = await createTestUser(agency, 'OWNER');
    workers = await createTestWorkers(agency, 10, { compliant: true });
    shift = await createTestShift(agency, { requiredCount: 5 });
  });

  afterAll(async () => {
    // Cleanup
    await cleanupDatabase();
  });

  describe('R-SA-01: Bulk Assignment API', () => {
    test('bulk assign 10 compliant workers to shift', async () => {
      const res = await request(app)
        .post(`/api/shifts/${shift.id}/assign-bulk`)
        .set('Authorization', `Bearer ${coordinatorToken}`)
        .send({ workerIds: workers.slice(0, 10).map(w => w.id), assignmentType: 'manual' });
      expect(res.status).toBe(200);
      expect(res.body.assigned.length).toBe(10);
      expect(res.body.skipped.length).toBe(0);
    });

    test('bulk assign mixed (5 compliant, 5 non-compliant)', async () => {
      // Setup: 5 workers with expired docs
      const nonCompliantWorkers = await createTestWorkers(agency, 5, { compliant: false, missingDocs: true });
      const allWorkers = [...workers.slice(0, 5), ...nonCompliantWorkers];
      
      const res = await request(app)
        .post(`/api/shifts/${shift.id}/assign-bulk`)
        .set('Authorization', `Bearer ${coordinatorToken}`)
        .send({ workerIds: allWorkers.map(w => w.id) });
      
      expect(res.status).toBe(200);
      expect(res.body.assigned.length).toBe(5);
      expect(res.body.skipped.length).toBe(5);
      expect(res.body.skipped[0]).toHaveProperty('reason');
      expect(res.body.skipped[0].reason).toMatch(/Missing|expired/i);
    });

    test('concurrent bulk assign same shift prevents duplicates', async () => {
      const worker = workers[0];
      const promises = [
        request(app).post(`/api/shifts/${shift.id}/assign-bulk`).set('Authorization', `Bearer ${coordinatorToken}`).send({ workerIds: [worker.id] }),
        request(app).post(`/api/shifts/${shift.id}/assign-bulk`).set('Authorization', `Bearer ${coordinatorToken}`).send({ workerIds: [worker.id] })
      ];
      const results = await Promise.all(promises);
      
      // One succeeds, one fails/returns in skipped
      const assignedCount = results.reduce((sum, r) => sum + r.body.assigned.length, 0);
      const skippedCount = results.reduce((sum, r) => sum + r.body.skipped.length, 0);
      
      // Either first succeeds (assigned=1, skipped=0) or second skipped (assigned=0, skipped=1)
      expect(assignedCount + skippedCount).toBe(1);
      
      // Verify only 1 ShiftAssignment row created
      const assignments = await prisma.shiftAssignment.findMany({ where: { shiftId: shift.id, workerId: worker.id } });
      expect(assignments.length).toBe(1);
    });
  });

  describe('R-SA-05: Worker Shift Confirmation', () => {
    test('worker confirms assigned shift', async () => {
      const assignment = await createTestAssignment(shift, workers[0]);
      
      const res = await request(app)
        .patch(`/api/worker-assignments/${assignment.id}`)
        .set('Authorization', `Bearer ${workerToken}`)
        .send({ action: 'confirm' });
      
      expect(res.status).toBe(200);
      expect(res.body.workerConfirmation).toBe('confirmed');
      expect(res.body.workerNote).toBeNull();
    });

    test('worker declines shift with reason', async () => {
      const assignment = await createTestAssignment(shift, workers[1]);
      const reason = 'Already committed elsewhere';
      
      const res = await request(app)
        .patch(`/api/worker-assignments/${assignment.id}`)
        .set('Authorization', `Bearer ${workerToken}`)
        .send({ action: 'decline', reason });
      
      expect(res.status).toBe(200);
      expect(res.body.workerConfirmation).toBe('declined');
      expect(res.body.workerNote).toBe(reason);
    });

    test('worker cannot confirm another worker\'s shift (403)', async () => {
      const assignment = await createTestAssignment(shift, workers[0]);
      const otherWorkerToken = generateWorkerToken(workers[1]);
      
      const res = await request(app)
        .patch(`/api/worker-assignments/${assignment.id}`)
        .set('Authorization', `Bearer ${otherWorkerToken}`)
        .send({ action: 'confirm' });
      
      expect(res.status).toBe(403);
    });
  });

  describe('R-SA-04 & R-SA-06: Email Notifications & Compliance Snapshot', () => {
    test('email sent when worker assigned', async () => {
      // Mock Resend or check FailedAlert queue
      const mockSendEmail = jest.spyOn(resendClient.emails, 'send').mockResolvedValue({ id: 'msg_123' });
      
      const res = await request(app)
        .post(`/api/shifts/${shift.id}/assign-bulk`)
        .set('Authorization', `Bearer ${coordinatorToken}`)
        .send({ workerIds: [workers[0].id] });
      
      // Give async email time to queue
      await new Promise(r => setTimeout(r, 100));
      
      expect(mockSendEmail).toHaveBeenCalledWith(expect.objectContaining({
        to: workers[0].email,
        subject: expect.stringContaining('[Shift Confirmed]')
      }));
    });

    test('compliance snapshot captured and immutable', async () => {
      const assignment = await createTestAssignment(shift, workers[0]);
      
      // Verify snapshot exists
      expect(assignment.complianceSnapshot).toBeTruthy();
      expect(assignment.complianceSnapshot.complianceScore).toBe(100);
      expect(assignment.complianceSnapshot.status).toBe('compliant');
      expect(assignment.complianceSnapshot.documents).toBeDefined();
    });
  });

  describe('R-SA-07: Shift Requirement Templates', () => {
    test('create shift requirement template', async () => {
      const res = await request(app)
        .post('/api/shift-requirements')
        .set('Authorization', `Bearer ${coordinatorToken}`)
        .send({
          templateName: 'Nursing',
          requiredDocuments: ['DBS', 'RTW', 'Training'],
          role: 'Nurse',
          description: 'Standard nursing shift requirements'
        });
      
      expect(res.status).toBe(201);
      expect(res.body.templateName).toBe('Nursing');
      expect(res.body.requiredDocuments).toHaveLength(3);
    });

    test('list templates for agency', async () => {
      const res = await request(app)

        .set('Authorization', `Bearer ${coordinatorToken}`);
      
      expect(res.status).toBe(200);
      expect(Array.isArray(res.body)).toBe(true);
      expect(res.body.length).toBeGreaterThan(0);
    });
  });

  describe('R-SA-08: Assignment Status Dashboard', () => {
    test('dashboard returns correct summary metrics', async () => {

      expect(res.body).toHaveProperty('summary');

    });

        where: { action: 'shift.assigned', entityId: res.body.assigned[0].id }

      expect(logs.length).toBeGreaterThan(0);

        where: { action: 'shift.assignment-confirmed', entityId: assignment.id }
      });
      
      expect(logs.length).toBeGreaterThan(0);

describe('Phase 8 End-to-End Workflow: Assign → Confirm → Audit Trail', () => {
  let agency, coordinator, shift, workers;

  beforeAll(async () => {
    agency = await createTestAgency();
    coordinator = await createTestUser(agency, 'OWNER');
    workers = await createTestWorkers(agency, 5, { compliant: true });
  });

  test('complete assignment workflow with audit trail', async () => {
    // Step 1: Create shift
    const shiftRes = await request(app)
      .post('/api/shifts')
      .set('Authorization', `Bearer ${coordinatorToken}`)
      .send({
        facilityName: 'St Mary\'s Hospital',
        shiftDate: '2026-06-05',
        startTime: '08:00',
        endTime: '16:00',
        role: 'Nursing Assistant',
        requiredCount: 3,
        notes: 'Standard shift, parking available'
      });
    
    expect(shiftRes.status).toBe(201);
    shift = shiftRes.body;
    
    // Step 2: Bulk assign workers
    const assignRes = await request(app)
      .post(`/api/shifts/${shift.id}/assign-bulk`)
      .set('Authorization', `Bearer ${coordinatorToken}`)
      .send({
        workerIds: workers.slice(0, 5).map(w => w.id),
        assignmentType: 'manual'
      });
    
    expect(assignRes.status).toBe(200);
    expect(assignRes.body.assigned.length).toBe(5);
    expect(assignRes.body.skipped.length).toBe(0);
    
    // Step 3: Verify assignments created
    const assignments = await prisma.shiftAssignment.findMany({
      where: { shiftId: shift.id }
    });
    expect(assignments.length).toBe(5);
    assignments.forEach(a => {
      expect(a.workerConfirmation).toBe('pending');
      expect(a.complianceSnapshot).toBeTruthy();
    });
    
    // Step 4: Verify emails sent
    const emailLogs = await prisma.auditLog.findMany({
      where: { action: 'email.sent', metadata: { shiftId: shift.id } }
    });
    expect(emailLogs.length).toBe(5);
    
    // Step 5: Verify audit logs for assignment
    const assignmentLogs = await prisma.auditLog.findMany({
      where: { action: 'shift.assigned', entityId: { in: assignments.map(a => a.id) } }
    });
    expect(assignmentLogs.length).toBe(5);
    
    // Step 6: Worker 1 confirms
    const worker1Token = generateWorkerToken(workers[0]);
    const confirmRes = await request(app)
      .patch(`/api/worker-assignments/${assignments[0].id}`)
      .set('Authorization', `Bearer ${worker1Token}`)
      .send({ action: 'confirm' });
    
    expect(confirmRes.status).toBe(200);
    expect(confirmRes.body.workerConfirmation).toBe('confirmed');
    
    // Step 7: Verify worker 1 confirmation logged
    const confirmLogs = await prisma.auditLog.findMany({
      where: { action: 'shift.assignment-confirmed', entityId: assignments[0].id }
    });
    expect(confirmLogs.length).toBe(1);
    
    // Step 8: Worker 2 declines with reason
    const worker2Token = generateWorkerToken(workers[1]);
    const declineRes = await request(app)
      .patch(`/api/worker-assignments/${assignments[1].id}`)
      .set('Authorization', `Bearer ${worker2Token}`)
      .send({ action: 'decline', reason: 'Scheduling conflict' });
    
    expect(declineRes.status).toBe(200);
    expect(declineRes.body.workerConfirmation).toBe('declined');
    expect(declineRes.body.workerNote).toBe('Scheduling conflict');
    
    // Step 9: Verify worker 2 decline logged
    const declineLogs = await prisma.auditLog.findMany({
      where: { action: 'shift.assignment-declined', entityId: assignments[1].id }
    });
    expect(declineLogs.length).toBe(1);
    expect(declineLogs[0].metadata.reason).toBe('Scheduling conflict');
    
    // Step 10: Verify complete audit trail
    const allLogs = await prisma.auditLog.findMany({
      where: { entityId: shift.id }
    });
    
    const actions = allLogs.map(l => l.action).sort();
    expect(actions).toContain('shift.created'); // From shift creation
    expect(actions).toContain('shift.assigned'); // From bulk assign
    expect(actions).toContain('shift.assignment-confirmed'); // From worker confirm
    expect(actions).toContain('shift.assignment-declined'); // From worker decline
    
    // Step 11: Verify dashboard shows correct metrics
    const dashboardRes = await request(app)
      .get(`/api/shifts/analytics/dashboard?shiftId=${shift.id}`)
      .set('Authorization', `Bearer ${coordinatorToken}`);
    
    expect(dashboardRes.status).toBe(200);
    const shiftMetrics = dashboardRes.body.shifts.find(s => s.id === shift.id);
    expect(shiftMetrics.assignedCount).toBe(5);
    expect(shiftMetrics.requiredCount).toBe(3);
    expect(shiftMetrics.fillRate).toBe(100); // 5 assigned, 3 required → 100% (capped)
    
    console.log('E2E workflow complete: ✓ Shift created ✓ Workers assigned ✓ Emails sent ✓ Worker confirmed ✓ Worker declined ✓ Audit trail logged');
  });
});
